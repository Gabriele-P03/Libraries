42 inline std::string getRootPath() noexcept{
43
44 std::string pathToFile = "";
45
46 //If the os is linux
47 #ifdef __linux__
48 char path[PATH_MAX];
49 ssize_t result = ::readlink("/proc/self/exe", path, sizeof(path)-1);
50 if(result != -1){
51 path[result] = '\0';
52 pathToFile = dirname(path);
53 }
54
55 #elif _WIN32
56 char path[PATH_MAX];
57
58 //When NULL is passed, directory of the exe is returned
59 HMODULE hmodule = GetModuleHandle(NULL);
60 if(hmodule != NULL){
61 GetModuleFileName(hmodule, path, sizeof(path));
62 }
63 pathToFile = path;
64 pathToFile = pathToFile.substr(0, pathToFile.find_last_of('\\'));
65 #else
66 #warning "Unrecognized O.S."
67 #endif
68
69 return pathToFile;
70 }

155 inline void readFile(std::fstream* file, char* &buffer, unsigned long &_size){
